                      function getNumbers() {
                        let startValue = document.getElementById("startValue").value;
                        let endValue = document.getElementById("endValue").value;
                      
                        //parse our numbers//
                        startValue = parseInt(startValue);
                        endValue = parseInt(endValue);
                        
                      
                        if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
                      
                          let numbers = generateNumbers(startValue,endValue);
                          
                          displayValues(numbers);
                          
                        } else {
                          Swal.fire({
                            backdrop: false,
                            title: "Oops!",
                            text: "You can only enter numbers",
                          });
                        }
                      }
                      
                      function generateNumbers(start,stop){
                        let numbers = [];
                      
                        for (let i = start; i <= stop; i++) {
                          numbers.push(i);    
                        }
                      
                        return numbers;
                      }
                      
                      function displayValues(numArray){
                          let results = document.getElementById("results")
                          results.innerHTML = "";

                        for (let index = 0; index < numArray.length; index++) {
                          let div = document.createElement("div");
                         let currentNumber = numArray[index];
                      
                          div.innerHTML = numArray[index];
                          div.classList.add("col");
                      
                          if(currentNumber % 2 == 0){
                            div.classList.add("odd");
                          }else{
                            div.classList.add("even");
                          }
                          results.appendChild(div);
                        }
                      
                      }

The code is structured in several functions

getNumbers()

Gets values input by the user.

generateNumbers()

If the number is between the numbers entered, raise it by 1

displayValues()

Displays even or odd numbers that occur between the numbers entered.
